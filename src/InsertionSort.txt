package com.Algorithm;

import java.util.*;

/**
 * 
 * @author Aditya
 *
 */

public class InsertionSort {
	/**
	 * 
	 * @param array
	 * @param length
	 */

	public static void sortStringarray(String array[], int length) {
		String flag = "";
		for (int i = 0; i < length; i++) {
			for (int j = i + 1; j < length; j++) {
				if (array[i].compareToIgnoreCase(array[j]) > 0) {
					flag = array[i];
					array[i] = array[j];
					array[j] = flag;
				}
			}
		}

	}

	/**
	 * 
	 * @param args
	 */

	/*
	 * this is the main method of our program
	 */

	public static void main(String[] args) {
		/*
		 * Scanner Object
		 */
		Scanner sc = new Scanner(System.in);

		System.out.println("Enter size of list to be sorted");

		int size = sc.nextInt();

		String[] arr = new String[size];

		System.out.println("Enter words for list that array will contain");

		for (int i = 0; i < size; i++) {
			arr[i] = sc.next();
		}
		sortStringarray(arr, arr.length);
		System.out.println("The sorted array is ");
		for (int i = 0; i < size; i++) {

			System.out.print(arr[i] + " ");
		}

	}

}
