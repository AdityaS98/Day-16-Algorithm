package com.AlgorithmProblems;

import java.util.*;
/**
 * 
 * @author DELL
 *
 */

public class TaskCompletion {
	/**
	 * 
	 * @param args
	 */
	/*
	 * This is the main method of our program
	 */

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);

		System.out.println("Enter the total time you have");
		/*
		 * Taking input of time
		 */

		double timeTaken = sc.nextDouble();

		System.out.println("Enter number of tasks");
		/*
		 * Taking Input of Tasks
		 */

		int tasks = sc.nextInt();

		double[] timerefuiredforTask = new double[tasks];

		System.out.println("Enter time in minutes for each task");

		for (int i = 0; i < tasks; i++) {
			timerefuiredforTask[i] = sc.nextDouble();
		}

		for (int i = 0; i < tasks; i++) {
			
			System.out.println("The folloeing task" + (i + 1) + " is completed at " + timeTaken + timerefuiredforTask[i]);
		}

	}

}
